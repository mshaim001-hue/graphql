# Tomorrow School API Configuration
# Copy this file to .env and fill in your credentials

# Option 1: Use JWT token directly (recommended)
TOMORROW_SCHOOL_JWT=your_jwt_token_here

# Option 2: Use username and password (will authenticate automatically)
TOMORROW_SCHOOL_USERNAME=your_username_or_email
TOMORROW_SCHOOL_PASSWORD=your_password

# Note: If you set TOMORROW_SCHOOL_JWT, the username/password will be ignored

# How to use:
# 1. Copy this file to .env
# 2. Fill in your credentials
# 3. Run: python3 audit_query.py
